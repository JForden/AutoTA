    def arg(self):
        if self.re >0:
            return math.atan(self.im/self.re)
        if self.re<0 and self.im >=0:
            return math.atan((self.im/self.re)+math.pi)
        if self.re<0 and self.im<0:
            return ((math.atan(self.im/self.re)-math.pi)
        if self.re ==0 and self.im >0:
            return(math.pi/2)
        if self.re ==0 and self.im<0:
            return((math.pi/2)*-1)
        if self.re==0 and self.im==0:
            raise ValueError
